bowtie2-build -f chr19.fa mm10bt2

brew install cpanminus; cpanm install XML::Simple

bowtie2 -x mm10bt2 input_ER4.fastq -S input_ER4.sam

macs2 callpeak -t CTCF_ER4.sam -c input_ER4.sam -g 61431566 -n ER4

macs2 callpeak -t CTCF_G1E.sam -c input_G1E.sam -g 61431566 -n G1E

bedtools intersect -a G1E_peaks.narrowPeak -b ER4_peaks.narrowPeak -v > diff_minus.bed
